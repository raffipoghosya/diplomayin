import numpy as np import numpy_financial as npf import pandas as pd # ── Define annual cash flows ── leasing_cf = np.array([-4230, 2470, 2670, 2870, 3070, 3270]) credit_cf = np.array([-9124, 2265, 2465, 2665, 2865, 3065]) # ── NPV at r = 10% ── r = 0.10 npv_leasing = npf.npv(r, leasing_cf) # → 6,506M AMD npv_credit = npf.npv(r, credit_cf) # → 834M AMD # ── Internal Rate of Return ── irr_leasing = npf.irr(leasing_cf) # → 42.70% irr_credit = npf.irr(credit_cf) # → 12.38% # ── Sensitivity sweep ── rates = np.arange(0.05, 0.26, 0.01) sensitivity = pd.DataFrame({ 'Rate': rates, 'NPV_Leasing': [npf.npv(r, leasing_cf) for r in rates], 'NPV_Credit': [npf.npv(r, credit_cf) for r in rates], }) # ── Comparison table ── result = pd.DataFrame({ 'Metric': ['NPV', 'IRR', 'Payback', 'ROI'], 'Leasing': [f'{npv_leasing:,.0f}', f'{irr_leasing*100:.2f}%', '2.47y', '49.9%'], 'Credit': [f'{npv_credit:,.0f}', f'{irr_credit*100:.2f}%', '4.69y', '22.5%'], }) print(result.to_string(index=False))
