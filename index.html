import numpy_financial as npf
import pandas as pd
import matplotlib.pyplot as plt

# ԼԻԶԻՆԳ — Cash Flows (Մլն Դրամ)
cf_L = [-4230, 1245.2, 1818.9,
         2448.0, 2638.7, 2798.7,
         2759.2, 2269.8]

# ՎԱՐԿ — Cash Flows (Մլն Դրամ)
cf_V = [-9124, 1000.8, 1641.8,
         2260.1, 2439.0, 2586.2,
         2532.8, 2517.6]

r = 0.10   # WACC Դիսկոնտի Դրույք

# NPV և IRR Հաշվարկներ
npv_L = npf.npv(r, cf_L)    # → 6,507 Մլն
irr_L = npf.irr(cf_L) * 100 # → 42.70%
npv_V = npf.npv(r, cf_V)    # → 834 Մլն
irr_V = npf.irr(cf_V) * 100 # → 12.38%

# ROI Հաշվարկներ
roi_L = (sum(cf_L[1:]) + cf_L[0]) / abs(cf_L[0]) * 100
roi_V = (sum(cf_V[1:]) + cf_V[0]) / abs(cf_V[0]) * 100

print(f"Լիզինգ → NPV={npv_L:.0f}M | IRR={irr_L:.2f}% | ROI={roi_L:.1f}%")
print(f"Վարկ   → NPV={npv_V:.0f}M | IRR={irr_V:.2f}% | ROI={roi_V:.1f}%")
# Արդյունքներ.
# Լիզինգ → NPV=6507M | IRR=42.70% | ROI=49.9%
# Վարկ   → NPV=834M  | IRR=12.38% | ROI=22.5%
